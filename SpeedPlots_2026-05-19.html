Load Data

trial_speeds <- read.csv("C:/Users/bankscai/OneDrive - Kennedy Krieger/Brooke and Caitlin/Dynamic Treadmill/Manuscript/CB Control Study/Data and Code/CBcontrols_TrialAvgSpeeds_2026-03-10.csv")
last30_speeds <- read.csv("C:/Users/bankscai/OneDrive - Kennedy Krieger/Brooke and Caitlin/Dynamic Treadmill/Manuscript/CB Control Study/Data and Code/CBcontrols_Last30AvgSpeeds_2026-03-24.csv")

trial_speeds <- trial_speeds %>% filter(Var1 !='GR0332' & Var1 != 'GR0333')
last30_speeds <- last30_speeds %>% filter(Var1 !='GR0332' & Var1 != 'GR0333')

colnames(trial_speeds) <- c('Subject','Event',seq(from=0,to=100,by=1))
colnames(last30_speeds) <- c('Subject','Event',seq(from=0,to=100,by=1))

trial_speeds_long <- pivot_longer(trial_speeds,cols=3:103,names_to="GaitCycle",values_to="Speed")
trial_speeds_long$GaitCycle <- as.numeric(trial_speeds_long$GaitCycle)
trial_speeds_long$Speed[is.nan(trial_speeds_long$Speed)] <- NA
last30_speeds_long <- pivot_longer(last30_speeds,cols=3:103,names_to="GaitCycle",values_to="Speed")
last30_speeds_long$GaitCycle <- as.numeric(last30_speeds_long$GaitCycle)
last30_speeds_long$Speed[is.nan(last30_speeds_long$Speed)] <- NA

last30_speeds_long$Event <- factor(last30_speeds_long$Event,levels=c("FastHS","SlowHS","AcceleratingHS","DeceleratingHS"))

estimation_rmse <- read.csv("C:/Users/bankscai/OneDrive - Kennedy Krieger/Brooke and Caitlin/Dynamic Treadmill/Manuscript/CB Control Study/Data and Code/rmse_max_table.csv")
estimation_rmse <- pivot_longer(estimation_rmse,cols=1:4,names_to="Event",values_to="RMSE")
estimation_rmse$RMSE[is.nan(estimation_rmse$RMSE)] <- NA
